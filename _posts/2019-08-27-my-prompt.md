# 수정중인 글입니다.

- Tested Environment : AIX, HP-UX, SunOS, Linux(RHEL, Ubuntu)

```sh
export PS1='$(printf "$(($(tput cols)+9-1))s\r%s\n", "infomation", "[$(hostname):$(whoami)]" $PWD
 > '
```
